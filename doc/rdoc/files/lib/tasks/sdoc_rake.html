# Rakefile require 'sdoc' # and use your RDoc task the same way you used it before require 'rdoc/task' # ensure this file is also required in order to use `RDoc::Task`

RDoc::Task.new do |rdoc|

rdoc.rdoc_dir = 'doc/rdoc' # name of output directory
rdoc.generator = 'sdoc' # explictly set the sdoc generator
rdoc.template = 'rails' # template used on api.rubyonrails.org
rdoc.main = 'README.md'
rdoc.rdoc_files.include('README.md', 'app/responses/', 'app/models/', 'app/speech_assets/', 'lib/', 'db/', 'config/')

end
