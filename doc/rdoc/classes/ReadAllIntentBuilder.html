def add
  add_slot(:Read, :READ) do |slot|
    slot.add_bindings(SlotSamples::READ_SLOTS)
  end
  add_slot(:Log, :LOG) do |slot|
    slot.add_bindings(SlotSamples::LOG_SLOTS)
  end

  add_intents(READ_ALL_INTENTS)
end
